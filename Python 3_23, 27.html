Immutable and Mutable

# id()
# 在 python 中，每一個 object 都會帶有一個獨一無二的 id
# 我們可以從 id 的變化判斷 object 是否為同一個，或是重新建了一個新的
def a(s):
    print(id(s))
    s += s
    print(id(s))

s = 9 # immutable，變更內容時必須重新建立 object
print(s, id(s))
a(s)
print(s, id(s))
def a(s):
    print(id(s))
    s += s
    print(id(s))

s = [9]   # mutable，可以直接修改 object 內容
print(s, id(s))
a(s)
print(s, id(s))

# fibo.py
def fib(n):    # write Fibonacci series up to n
    a, b = 0, 1
    while b < n:
        print b,
        # 1    | 1    |
        a, b = b, a+b
        # 1, 1 | 1, 2 |

def fib2(n):   # return Fibonacci series up to n
    result = []
    a, b = 0, 1
    while b < n:
        result.append(b)
        a, b = b, a+b
    return result
