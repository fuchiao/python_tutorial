# 2) 計算 a, a+1, a+2, ..., b 的總和
# 2-1) 迴圈解
def summ(a, b):
    pass
# 3-2) 遞迴解
def summ(a, b):
    if a >= b:
        return b
    else:
        return a + summ(a+1, b)

print(summ(3, 5))
print(summ(1, 4))
# 3) 計算第 n 位費波納西數列：1, 1, 2, 3, 5, 8, 13, ...
# 3-1) 迴圈解
def fib(n):
    pass
print(fib(5))
# 3-2) 遞迴解
def fib(n):
    if n <= 2:
        return 1
    else:
        return fib(n-1)+fib(n-2)
        
print(fib(1))
print(fib(2))
print(fib(3))
print(fib(5))

類別

  • 名詞說明
    • class
      • 定義類別
      • 底下通常會包含多個 member function
    • instance
      • 根據 class 所建立的實體
      • 建立時會執行 __init__()，通常會在此初始化 instance variable 
class Dog:
    ''' a class object for dogs '''
    def __init__(self, name):
        self.name = name
        self.tricks = []

    def add_trick(self, trick):
        self.tricks.append(trick)
        
d = Dog('Fido')
e = Dog('Buddy')
d.add_trick('roll over')
e.add_trick('play dead')
print(d.tricks)
print(e.tricks)
print(d.__doc__)    # 印出類別說明字串 docstring

class Dog:
    def __init__(self, name):
        self.name = name

    def eat(self):
        print(self.name+' eat something')


class BigDog(Dog):
    def eat(self):
        print(self.name+' eat a lot of things')


x = Dog('Obama')
y = BigDog('Trump')
x.eat()
y.eat()
